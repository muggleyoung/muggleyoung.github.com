<算法与数据结构>第四次上机报告

题目:使用哈希表查找单词

题目补充说明:

给定一个英文段落(单词个数小于200),利用hash表(表长最大20)统计单词出现的频度,并能根据要求显示出给定单词在段落中出现的位置.和不同单词的出现频度.其中hash函数使用除留余数法构造,使用链地址法进行冲突处理.

一概要设计:

1基本数据结构及操作:

//单词位置信息结构体;

typedef struct locate

{    

       int pos;                                     //位置链表的当前长度;

       struct locate *place;

}locate;                                                    //存储单词出现次数和出现位置的结构体;

//创建单词结构体;

typedef struct word

{

       sword letter;                                      //单词中的每一个字母组成的数组;

       int frequency;                                    //单词的使用频率;

       struct locate *place;                                   //指向单词的位置链表;

       struct word *next;                              //指向下一节点的指针;

}word,W[MAXSIZE];

//创建哈希表存储结构的结构体;

typedef struct

{

       struct word *next;                              //哈希表中该地址指向的搜索链表首结点;

}hashtable[MAXHASHNUM];

 

基本操作定义:

void inithash(hashtable h,int len)                 //初始化hash表;

void initword(word *&w,locate *&p)         //初始化单词的结点;

int hashkey(sword s)                              //定义hash函数;

void value(sword &s1,sword s2)                //对字符数组赋值;

int compare(sword s1,sword s2)                      //比较两个字符串是否相等;

word *exist(sword s,hashtable h,int num)   //判断hash表中是否已存在s;

void creathtable(hashtable h,sword s,int wordnum) //建立哈希表;

bool IsLetter(char temp)                                   //判断是不是字母;

void search(sword s,hashtable h)             //在hash表中查找,并打印信息;

void output(hashtable h)                          //输出文章中的所有单词信息;

二详细设计

//初始化hash表;

void inithash(hashtable h,int len)           //len为表长;

{

       int i;

       for(i=0; i<len; i++)

       {

              h[i].next=NULL;                                //指向链表的指针为空;

       }

}

 

//初始化单词的结点;

void initword(word *&w,locate *&p)

{

       w=(word *)malloc(sizeof(word));       

       p=(locate *)malloc(sizeof(locate));       //分配空间;

       w->frequency=0;                                      //初始频率为0;

       w->place=NULL;

       w->next=NULL;                                       //指针均为空;

}

 

//定义hash函数;

int hashkey(sword s)                                 //以每个单词的前两个字母的asc码值之和与20求余为hash函数值;

{

       int key;

       if(s[1]!=’\0’)                                      //单词的字母个数>=两个字母时;

              key=(s[0]+s[1])%20;

       else                                                   //只有一个字母的单词;

              key=s[0]%20;

       return key;

}

//对字符数组赋值;

void value(sword &s1,sword s2)                //将字符串s1赋给s2;

{

       int i;

       for(i=0;s2[i]!=’\0’;i++)                 //字符串的终止是‘\0’;

       {

              s1[i]=s2[i];

       }

       s1[i]=’\0’;

}

//比较两个字符串是否相等;

int compare(sword s1,sword s2)

{

       int i=0;

       while( i<20 && s1[i]!=’\0’ && s2[i]!=’\0’)

       {

              if(s1[i]!=s2[i])                            //不相等时返回0;

                     return 0;

              i++;

       }

   if(s1[i]==’\0’ && s2[i]==’\0’)         //相等时返回1;

              return 1;

       else

              return 0;

}
