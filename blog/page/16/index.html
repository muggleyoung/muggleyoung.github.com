Quick Sort 快排：

 

#include <fstream.h>
#include<math.h>

int partition(int A[], int head,int tail)
{
    int key = A[head];
    int i,j,t;
    i = head;
    j = head+1;
    while (j<=tail)
    {
        if (key>A[j])
        {
            i = i+1;
            t = A[j];
            A[j] = A[i];
            A[i] = t;
        }
        j = j+1;
    }
    A[head] = A[i];
    A[i] = key;
    return i;
}

void quicksort(int A[], int head,int tail)
{
    int r;
    if(head<tail)
    {
        r = partition(A,head,tail);
        quicksort(A,head,r);
        quicksort(A,r+1,tail);
    }
}

void main()
{
    ifstream infile;
    ofstream outfile (“result.txt”);
    infile.open(“seq.txt”);
    int n = 0;
    int A[100];
    while (!infile.eof())
    {
        infile>>A[n];
        n++;
    }
    quicksort(A,0,n-1);
    for (int i=0;i<n;i++)
    {
        outfile<<A[i]<<” “;
    }
}

MergeSort

merge sort 应该叫归并排序之类的吧

好久不用C，怎样把数组作为参数传递这个问题，着实困扰我好久

一直暴力的直接把数组声明为全局变量，不过用vector倒是个办法

今天看书，才想起来该怎样做

老啦，笨啦~

 

#include <fstream.h>
#include<math.h>

void merge(int A[], int head,int tail,int mid,int n)
{
    int i;
    int j;
    int k;
    int L[50];
    int R[50];
    for (i=0;i<mid-head+1;i++)
    {
        L[i] = A[head+i];
    }
    for (j=0;j<tail-mid;j++)
    {
        R[j] = A[mid+j+1];
    }
    i=0;
    j=0;
    for (k=head;k<=tail;k++)
    {
        if (L[i]>=R[j])
        {
            A[k] = L[i];
            i=i+1;
        }
        else
        {
            A[k] = R[j];
            j = j+1;
        }
    }
}

void mergesort(int A[], int head,int tail,int n)
{
    if (head<tail)
    {
        int mid;
        mid = ceil((tail+head)/2);
        mergesort(A,head,mid,n);
        mergesort(A,mid+1,tail,n);
        merge(A,head,tail,mid,n);
    }
}

void main()
{
    ifstream infile;
    ofstream outfile (“result.txt”);
    infile.open(“seq.txt”);
    int n = 0;
    int A[100];
    while (!infile.eof())
    {
        infile>>A[n];
        n++;
    }
    mergesort(A,0,n-1,n);
    for (int i=0;i<n;i++)
    {
        outfile<<A[i]<<” “;
    }
}
