冒泡排序：

#include <fstream>
#include <math.h>
using namespace std;

void BubbleSort(int A[],int n)
{
    for(int j=1;j!=n;j++)
    {
        for (int i=0;i!=(n-j);i++)
        {
            if(A[i]>A[i+1])
            {
                int t = A[i];
                A[i] = A[i+1];
                A[i+1] = t;
            }
        }
    }
}

void main()
{
    ifstream infile;
    ofstream outfile(“result.txt”);
    infile.open(“seq.txt”);
    int A[100];
    int n=0;
    while(!infile.eof())
    {
        infile>>A[n];
        n++;
    }
    BubbleSort(A,n);
    for (int i=0;i!=n;i++)
    {
        outfile<<<” “;
    }
}

QuickSort

Sep 20th, 2010

为毛我死活学不会计算时间复杂度~

太挫败了~

 

Quick Sort 快排：

 

#include <fstream.h>
#include<math.h>

int partition(int A[], int head,int tail)
{
    int key = A[head];
    int i,j,t;
    i = head;
    j = head+1;
    while (j<=tail)
    {
        if (key>A[j])
        {
            i = i+1;
            t = A[j];
            A[j] = A[i];
            A[i] = t;
        }
        j = j+1;
    }
    A[head] = A[i];
    A[i] = key;
    return i;
}

void quicksort(int A[], int head,int tail)
{
    int r;
    if(head<tail)
    {
        r = partition(A,head,tail);
        quicksort(A,head,r);
        quicksort(A,r+1,tail);
    }
}

void main()
{
    ifstream infile;
    ofstream outfile (“result.txt”);
    infile.open(“seq.txt”);
    int n = 0;
    int A[100];
    while (!infile.eof())
    {
        infile>>A[n];
        n++;
    }
    quicksort(A,0,n-1);
    for (int i=0;i<n;i++)
    {
        outfile<<A[i]<<” “;
    }
}
