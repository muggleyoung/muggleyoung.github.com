从Insert Sort开始，跟排序杠上了

 

读入文件seq.txt，里面存了好几个整数，具体数目不详；

输出result.txt，写入排序后的数。

算法：insert sort  插入排序

 

#include <fstream.h>
void main()
{
    ifstream infile;
    ofstream outfile (“result.txt”);
    int key;
    infile.open(“seq.txt”);
    int n = 0;
    int A[100];
    while (!infile.eof())
    {
        infile>>A[n];
        n++;
    }
    int i,j;
    for (j=1;j<n;j++)
    {
        key = A[j];
        for (i=j-1;i>=0;i–)
        {
            if (A[i]>key)
            {
                A[i+1] = A[i];
                A[i] = key;
            }
        }
    }
    for (i=1;i<n;i++)
    {
        outfile<<A[i]<<” “;
    }
}
