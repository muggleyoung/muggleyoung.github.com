Quick Sort 快排：

 

#include <fstream.h>
#include<math.h>

int partition(int A[], int head,int tail)
{
    int key = A[head];
    int i,j,t;
    i = head;
    j = head+1;
    while (j<=tail)
    {
        if (key>A[j])
        {
            i = i+1;
            t = A[j];
            A[j] = A[i];
            A[i] = t;
        }
        j = j+1;
    }
    A[head] = A[i];
    A[i] = key;
    return i;
}

void quicksort(int A[], int head,int tail)
{
    int r;
    if(head<tail)
    {
        r = partition(A,head,tail);
        quicksort(A,head,r);
        quicksort(A,r+1,tail);
    }
}

void main()
{
    ifstream infile;
    ofstream outfile (“result.txt”);
    infile.open(“seq.txt”);
    int n = 0;
    int A[100];
    while (!infile.eof())
    {
        infile>>A[n];
        n++;
    }
    quicksort(A,0,n-1);
    for (int i=0;i<n;i++)
    {
        outfile<<A[i]<<” “;
    }
}
