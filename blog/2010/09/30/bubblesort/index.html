BubbleSort

Sep 30th, 2010

冒泡排序：

#include <fstream>
#include <math.h>
using namespace std;

void BubbleSort(int A[],int n)
{
    for(int j=1;j!=n;j++)
    {
        for (int i=0;i!=(n-j);i++)
        {
            if(A[i]>A[i+1])
            {
                int t = A[i];
                A[i] = A[i+1];
                A[i+1] = t;
            }
        }
    }
}

void main()
{
    ifstream infile;
    ofstream outfile(“result.txt”);
    infile.open(“seq.txt”);
    int A[100];
    int n=0;
    while(!infile.eof())
    {
        infile>>A[n];
        n++;
    }
    BubbleSort(A,n);
    for (int i=0;i!=n;i++)
    {
        outfile<<<” “;
    }
}
