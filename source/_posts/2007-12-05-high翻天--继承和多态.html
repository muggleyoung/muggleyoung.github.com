完成目标:

通过实践体会C#的继承机制和多态的实现.

问题描述:

相比之下,由于对于运动类服装的销售了解更清楚些,放弃了关于图书的作业,自己重新描述了这个问题.先定义一个merchandise(商品)类,然后sportsware (运动装)继承merchandise, clothing和bag是sportsware的子类.在student类中定义虚函数 display()显示商品基本信息.而在子类, clothing, bag中分别重新定义display()显示其特有信息.

 源码:

using System;

using System.Collections.Generic;

using System.Text;

 

class merchandise

{

    protected string name; //商品名

    protected int number;//商品货号

    protected int price;//产品价格

    protected string date;//出厂日期

    protected merchandise()

    {

    }

    protected merchandise(string name, int number, int price, string date)

{

        this.name = name;

        this.number = number;

        this.price = price;

        this.date = date;

    }

    public string Name

    {

        get

        {

            return name;

        }

        set

        {

            name = value;

        }

    }

    public int Number

    {

        get

        {

            return number;

        }

        set

        {

            number = value;

        }

    }

    public int Price

    {

        get

        {

            return price;

        }

        set

        {

            price = value;

        }

    }

    public string Date

    {

        get

        {

            return date;

        }

        set

        {

            date = value;

        }

    }

}

class sportsware : merchandise//运动装

{

    private string material;//材质

    private string color;//颜色

 

    public sportsware()

{

   }

    public sportsware(string name, int number, int price, string date)

        : base(name, number, price, date)

{

}

    public sportsware(string name, int number, int price, string date, string material, string color)

       : base(name, number, price, date)

    {

        this.material = material;

        this.color = color;

    }

    public string Material

    {

        get

        {

            return material;

        }

        set

        {

            material = value;

        }

    }

    public string Color

    {

        get

        {

            return color;

        }

        set

        {

            color = value;

        }

    }   

    public virtual void display()

   {

        Console.WriteLine(" name: {0} \n number: {1} \n price: {2} \n date: {3} ",             Name, Number, Price, Date);

     Console.WriteLine(" material: {0} \n color: {1} ", Material, Color);

   }

}

class clothing : sportsware

{

private char sex;//男式或女式

private string size;//尺码

clothing()

{

 }

public clothing(string name, int number, int price, string date)

      : base(name, number, price, date)

{

}

public clothing(string name, int number, int price, string date, string material, string color)

      : base(name, number, price, date, material, color)

{

}

public clothing(string name, int number, int price, string date, string material, string color,char sex,string size)

       : base(name, number, price, date, material, color)

{

this.sex = sex;

    this.size = size;

 }

 public override void display()

{

    Console.WriteLine("This is the information about sports clothing:");

    base.display();

    Console.WriteLine("sex : {0}\n size : {1}", sex, size);

 }

}

class bag : sportsware

{

    private string type;//包的类型,双肩,单肩或腰包...

    public bag()

    {

    }

    public bag(string name, int number, int price, string date)

        : base(name, number, price, date)

    {

    }

    public bag(string name, int number, int price, string date, string material, string color)

        : base(name, number, price, date, material, color)

    {

    }

    public bag(string name, int number, int price, string date, string material, string color, string type)

        : base(name, number, price, date, material, color)

    {

        this.type = type;

    }

    public override void display()

    {

        Console.WriteLine("This is the information about sports bag:");

        base.display();

        Console.WriteLine("type : {0} ",type);

    }

}
